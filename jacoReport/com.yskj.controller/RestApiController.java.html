    @RequestMapping(value = {"/deleteapibyid/{apiid}"},method = RequestMethod.DELETE)
    public String delete(@PathVariable("apiid") String ids) {
        String[] idss = ids.split(",");
        for (String id : idss) {
            int idint = Integer.parseInt(id);
            apiService.delete(idint);
        }

        return "0";
    }
